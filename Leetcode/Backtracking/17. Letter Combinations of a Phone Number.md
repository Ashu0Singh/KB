[Question](https://leetcode.com/problems/letter-combinations-of-a-phone-number/)

### Optimal Solution

```java
class Solution {
    String[] phone = new String[]{"", "", "abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz" };

    public List<String> letterCombinations(String digits) {

        List<String> ans = new ArrayList<>();

        if (digits == null || digits.length() == 0) {
            return ans;
        }

        helper(ans, digits.toCharArray(), 0, new StringBuilder());

        return ans;
    }

    void helper(List<String> ans, char[] digits, int index, StringBuilder sequence){
        if(sequence.length() == digits.length || index >= digits.length) {
            ans.add(sequence.toString());
            return;
        }

        int number = Character.getNumericValue(digits[index]);

        for(char ch : phone[number].toCharArray()){
            sequence.append(ch);
            helper(ans,digits,index + 1, sequence);
            sequence.deleteCharAt(sequence.length() - 1);
        }

    }
}
```

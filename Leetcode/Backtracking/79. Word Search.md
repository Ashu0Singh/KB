### Optimal Solution

```java
class Solution {
    public boolean exist(char[][] board, String word) {
        for(int i  = 0; i < board.length; i++){
            for(int j = 0; j < board[0].length; j++){
                boolean isPath = backtrack(board, i, j, new boolean[board.length][board[0].length], word.toCharArray(), 0);
                if(isPath) return true;
            }
        }
        return false;
    }

    boolean backtrack(char[][] board, int r, int c, boolean[][] visited, char[] word, int idx) {
	    if (idx == word.length) return true;
	
	    if (r < 0 || c < 0 || r >= board.length || c >= board[0].length ||
	        visited[r][c] || board[r][c] != word[idx]) {
	        return false;
	    }
	
	    visited[r][c] = true;
	
	    boolean found =
	        backtrack(board, r + 1, c, visited, word, idx + 1) ||
	        backtrack(board, r - 1, c, visited, word, idx + 1) ||
	        backtrack(board, r, c + 1, visited, word, idx + 1) ||
	        backtrack(board, r, c - 1, visited, word, idx + 1);
	
	    visited[r][c] = false;
	
	    return found;
	}
}
```

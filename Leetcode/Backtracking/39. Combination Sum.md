### Optimal Approach

So here since we can have an element as many time as we want, what we will do is add a candidate to the list and then recursively call the helper function from current element to the end of list. At ever step we will add the current num of I and then remove it so that we can cover both the options, if a num was considered or not. 

```java
class Solution {
    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        List<List<Integer>> answer = new ArrayList<>();

        helper(new ArrayList<>(), answer, candidates, target, 0);

        return answer;
    }
    void helper(List<Integer> subset, List<List<Integer>> answer, int[] candidates, int target, int index){
        if(target < 0 || index >= candidates.length) return;
        if(target == 0){
            answer.add(new ArrayList<>(subset));
        } else {
            for( int i = index; i < candidates.length; i++){
                subset.add(candidates[i]);
                
                helper(subset, answer ,candidates, target - candidates[i], i);

                subset.remove(subset.size() - 1);
            }
        }
    }
}
```

class Solution {
    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        List<List<Integer>> answer = new ArrayList<>();

        helper(new ArrayList<>(), answer, candidates, target, 0);

        return answer;
    }
    void helper(List<Integer> subset, List<List<Integer>> answer, int[] candidates, int target, int index){
        if(target < 0 || index >= candidates.length) return;
        if(target == 0){
            answer.add(new ArrayList<>(subset));
        } else {
            subset.add(candidates[index]);
            helper(subset, answer,candidates, target - candidates[index], index);

            subset.remove(subset.size() - 1);
            helper(subset, answer,candidates, target, index + 1);
        }
    }
}

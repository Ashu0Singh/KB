### Optimal Solution

[[207. Course Schedule]] understand this and you'll easily understand the current implementation.

```java
class Solution {
    int index = 0;
    public int[] findOrder(int numCourses, int[][] prerequisites) {
        ArrayList<Integer>[] map = new ArrayList[numCourses];
        int[] path = new int[numCourses];

        for(int i = 0; i < numCourses; i++){
            map[i] = new ArrayList<>();
        }

        for(int[] edge : prerequisites){
            map[edge[0]].add(edge[1]);
        }

        for(int i = 0; i < numCourses; i++){
            if(!dfs(map, i, new boolean[numCourses], path)) return new int[0];
        }

        return path;
    }
    boolean dfs(ArrayList<Integer>[] map, int i, boolean[] visited, int[] path){
        if(map[i] == null) return true;
        if(visited[i]) return false;

        visited[i] = true;

        boolean answer = true;
        for(int edge : map[i]){
            answer = answer && dfs(map, edge, visited, path);
        }

        map[i] = null;
        path[index++] = i;
        visited[i] = false;
        return answer;
    }
}
```

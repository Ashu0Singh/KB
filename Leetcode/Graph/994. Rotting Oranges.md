### Optimal Solution

What we're gonna do here is basically identify all the rotting oranges initially, and then start a BFS traversal on it using a queue. For every rotting orange check for fresh adjacent orange and then if you find any, push to queue. To make sure you're going iteration wise always get the current size of the queue and only poll that many times. Since you're gonna be inserting to the same queue, you don't want to do it all in one iteration.

```java
class Solution {
    class Pair {
        int row;
        int col;
        Pair(int row, int col){
            this.row = row;
            this.col = col;
        }
        public String toString() {
            return "Row: " + row + ", Col: " + col;
        }
    }
    Queue<Pair> q = new LinkedList<>();
    public int orangesRotting(int[][] grid) {
        int count = -1;
        int m = grid.length;
        int n = grid[0].length;

        //Adding the initial state to the queue
        for(int i = 0; i < grid.length; i++){
            for(int j = 0; j < grid[0].length; j++){
                if(grid[i][j] == 2){
                    q.add(new Pair(i,j));
                }
            }
        }

        while(!q.isEmpty()){
            int size = q.size();

            while(size > 0){

                Pair top = q.poll();
                grid[top.row][top.col] = -1;

                if(top.row + 1 < m && grid[top.row + 1][top.col] == 1) {
                    grid[top.row + 1][top.col] = 2;
                    q.add(new Pair(top.row + 1, top.col));
                }
                if (top.col + 1 < n && grid[top.row][top.col + 1] == 1){
                    grid[top.row][top.col + 1] = 2;
                    q.add(new Pair(top.row, top.col + 1));
                }
                if(top.row - 1 >= 0 && grid[top.row - 1][top.col] == 1) {
                    grid[top.row - 1][top.col] = 2;
                    q.add(new Pair(top.row - 1, top.col));
                }
                if (top.col - 1 >= 0 && grid[top.row][top.col - 1] == 1){
                    grid[top.row][top.col - 1] = 2;
                    q.add(new Pair(top.row, top.col - 1));
                }

                size--;
            }
            count++;
        }

        for(int i = 0; i < grid.length; i++){
            for(int j = 0; j < grid[0].length; j++){
                if(grid[i][j] == 1){
                    return -1;
                }
            }
        }

        if(count == -1) return 0;

        return count;
    }
}
```

### Optimal Solution

```java
class Solution {
    public boolean isAnagram(String s, String t) {
        if(s.length() != t.length()) return false;

        char[] sChar = s.toCharArray();
        char[] tChar = t.toCharArray();

        int[] map = new int[26];

        for(int i = 0; i < sChar.length; i++){
            map[sChar[i] - 'a']++;
            map[tChar[i] - 'a']--;
        }

        for(int count : map) {
            if ( count != 0) return false;
        }

        return true;
    }
}
```

[50. Pow(x, n)](https://leetcode.com/problems/powx-n/)

### Optimal Solution
```java
class Solution {
    public double myPow(double x, int n) {
        long pow = n;
        double ans = 1;
        if(pow < 0) pow = pow * -1;
        while(pow > 0){
            if(pow % 2 == 0){
                x = x * x;
                pow = pow / 2;
            }
            else {
                ans = ans * x;
                pow--;
            }
        }
        if(n < 0) ans = 1/ans;
        return ans;
    }
}
```

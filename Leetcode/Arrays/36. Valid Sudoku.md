[Question](https://leetcode.com/problems/valid-sudoku/description/)

### Optimal Solution

The best way is to iterate through every row and column to check for having the same number along with having a grid to check whether we have any duplicates or not?

```java
class Solution {
    public boolean isValidSudoku(char[][] board) {
        HashSet<Integer>[][] grids = new HashSet[3][3];

        for(int i = 0; i < 3; i++){
            for(int j = 0; j < 3; j++){
                grids[i][j] = new HashSet<Integer>();
            }
        }
        for(int i = 0; i < 9; i++){
            HashSet<Integer> row = new HashSet<>();
            HashSet<Integer> col = new HashSet<>();
            for(int j = 0; j < 9; j++){
                int currRow = board[i][j] - '0';
                int currCol = board[j][i] - '0';

                HashSet<Integer> currGrid = grids[i/3][j/3];
                if(currRow > 0 && currRow < 10 && (!row.add(currRow) || !currGrid.add(currRow))) return false;
                if(currCol > 0 && currCol < 10 && !col.add(currCol)) return false;
            }
        }

        return true;
    }
}
```

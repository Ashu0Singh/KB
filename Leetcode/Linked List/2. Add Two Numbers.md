### Optimal Approach

Use the merge function and modify it to accommodate for the question

```java
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        int carry = 0;
        ListNode answer = new ListNode();
        ListNode ansHead = answer;
        while(l1 != null && l2 != null ){
            int sum = carry + l1.val + l2.val;
            carry = sum / 10;
            answer.next = new ListNode(sum % 10);
            answer = answer.next;
            l1 = l1.next;
            l2 = l2.next;
        }
        while(l1 != null){
            int sum = carry + l1.val;
            carry = sum / 10;
            answer.next = new ListNode(sum % 10);
            answer = answer.next;
            l1 = l1.next;
        }
        while(l2 != null ){
            int sum = carry + l2.val;
            carry = sum / 10;
            answer.next = new ListNode(sum % 10);
            answer = answer.next;
            l2 = l2.next;
        }
        if(carry > 0) answer.next = new ListNode(carry);

        return ansHead.next;
    }
}
```

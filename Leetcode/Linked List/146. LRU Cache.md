### Optimal Soultion 

Here is what we're doing here for LRU cache, first we will have HashMap to store every node associated with a key and a doubly linkedlist to be able to traverse through it easily. We will start with initialising a head and a end pointer of the DLL and then whenever a key is put, insert it to the first place, whenever a key is fetched move it to the first place. 

Once these two things are sorted we will make sure that the edge cases make sense. When we run out of capacity remove the last node in the DLL and move the end pointer. If a pre-existing key is put, then just change the node value and use the get call to move it to first place. 

```java
class LRUCache {
    int cap;
    HashMap<Integer, Node> map;

    class Node {
        int key;
        int val;
        Node next;
        Node prev;

        Node(int key, int val, Node next, Node prev){
            this.key = key;
            this.val = val;
            this.next = next;
            this.prev = prev;
        }
    }

    Node head;
    Node end;

    public LRUCache(int capacity) {
        this.cap = capacity;
        map = new HashMap<>();
        head = new Node(-1, -1, null, null);
        end = head;
    }

    public int get(int key) {
        if(map.containsKey(key)) {

            Node latest = map.get(key);

            if(latest.prev != null) latest.prev.next = latest.next;
            if(latest.next != null) latest.next.prev = latest.prev;

            if(latest == end) end = latest.prev;

            latest.next = head.next;
            latest.prev = head;

            if(head.next != null) head.next.prev = latest;

            head.next = latest;

            if(end == head) end = latest;

            return latest.val;
        }
        return -1;
    }

    public void put(int key, int value) {

        if(map.containsKey(key)) {
            Node node = map.get(key);
            node.val = value;
            get(key);
            return;
        }

        if(cap == 0) {
            map.remove(end.key);
            if(end.prev != null) {
                end.prev.next = null;
                end = end.prev;
            }
            cap++;
        }

        Node curr = new Node(key, value, head.next, head);
        if(head.next != null) head.next.prev = curr;
        head.next = curr;
        if(end == head) end = curr;
        map.put(key, curr);
        cap--;
    }
}
```

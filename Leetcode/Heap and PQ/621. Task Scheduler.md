### Better Solution

```java
public class Task{
    int taskCount;
    int cooldown;

    public Task(int taskCount, int cooldown){
        this.taskCount = taskCount;
        this.cooldown = cooldown;
    }

    public int getTaskCount() { return this.taskCount; }
    public int getCooldown() { return this.cooldown; }

    //For Debugging
    @Override
    public String toString() {
        return "(" + taskCount + ", " + cooldown + ")";
    }
}

class Solution {
    public int leastInterval(char[] tasks, int n) {
        HashMap<Character,Integer> map = new HashMap<>();
        PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
        Queue<Task> taskQueue = new LinkedList<>();
        int count = 0;

        for(char task : tasks) map.put(task, map.getOrDefault(task, 0) + 1);

        for(Integer value : map.values()) pq.offer(value);

        while(!pq.isEmpty() || !taskQueue.isEmpty()){
            count++;

            if(!pq.isEmpty()){
                int curr = pq.poll() - 1;
                if(curr > 0){
                    taskQueue.offer(new Task(curr, count + n));
                }
            }

            if(!taskQueue.isEmpty() && taskQueue.peek().getCooldown() == count){
                pq.offer(taskQueue.poll().getTaskCount());
            }
        }
        return count;
    }
}
```

### Optimal Solution 

```java
class Solution {
    public int leastInterval(char[] tasks, int n) {
        int[] freq = new int[26];

        for (char c : tasks) {
            freq[c - 'A']++;
        }

        int maxFreq = 0;
        int maxCount = 0;

        for (int f : freq) {
            if (f > maxFreq) {
                maxFreq = f;
                maxCount = 1;
            } else if (f == maxFreq) {
                maxCount++;
            }
        }

        int partCount = maxFreq - 1;
        int partLength = n + 1;

        int minTime = partCount * partLength + maxCount;

        return Math.max(tasks.length, minTime);
    }
}
```

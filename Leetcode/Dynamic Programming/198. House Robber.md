### Optimal Solution

```java
class Solution {
    public int rob(int[] nums) {
        for(int i = 1; i < nums.length; i++){
            int robNow = nums[i];
            if(i > 1) robNow += nums[i - 2];
            if(nums[i - 1] > robNow) robNow = nums[i-1]; 
            nums[i] = robNow;
        }
        return nums[nums.length - 1];
    }
}
```
